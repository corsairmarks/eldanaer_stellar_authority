# Hibernating Lithoids are generated on lithoid homeworlds at game-start
d_hibernating_zdremites = {
	icon = d_hibernating_lithoids
	time = 360
	is_for_colonizable = yes
	category = deposit_cat_rare_blocker
	potential = { always = no }
	on_cleared = {
		if = {
			limit = {
				any_owned_pop = {
					species = { home_planet = { is_planet = PREVPREVPREV } }
					is_lithoid = yes
				}
			}
			random_owned_pop = {
				limit = {
					species = { home_planet = { is_planet = PREVPREVPREV } }
					is_lithoid = yes
				}
				ROOT = {
					create_pop = {
						species = PREV
					}
				}
			}
		}
		else_if = {
			limit = { any_owned_pop = { is_lithoid = yes } }
			random_owned_pop = {
				limit = { is_lithoid = yes }
				ROOT = {
					create_pop = {
						species = PREV
					}
				}
			}
		}
		else = {
			random_owned_pop = {
				ROOT = {
					create_pop = {
						species = PREV
					}
				}
			}
		}
	}
	resources = {
		category = deposit_blockers
		cost = {
			minerals = 1000
		}
	}
	ai_weight = {
		weight = 1
		modifier = {
			factor = 2
			from = {
				is_homeworld = yes
			}
		}
	}
}
